```typescript
export interface CategoryConfig {
  id: CategoryId;
  displayName: string;
  description: string;
  icon: string;
  unit: string;
  tiers: TierLevel[];  // Must be sorted by threshold descending
}

export interface TiersConfig {
  version: string;
  categories: CategoryConfig[];
}
```

Create src/lib/tiers-config.ts as the config loader:

```typescript
import tiersJson from '@/config/tiers.json';
import type { TiersConfig, CategoryConfig, CategoryId } from './tiers-types';

// Type assertion for build-time validation
export const tiersConfig = tiersJson as TiersConfig;

// Helper to get category config by ID
export function getCategoryConfig(id: CategoryId): CategoryConfig | undefined {
  return tiersConfig.categories.find(cat => cat.id === id);
}
```
